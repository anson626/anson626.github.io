#!/bin/bash

platForm=(1038 84 786 368 696 617 51 58 wns tiantian libo jinsha n01)
jar_name=(activity-api business-api admin-api game-betting gateway job-api special-api h5-admin)
action_name=(update exit_sys)

show_platForm(){
    count1=1
    for i in ${platForm[*]}
    do
        echo -e "\033[44m $count1>   $i \033[0m"
        let ++count1
    done
    read -e -p '请输入要操作的平台:' num1
}

show_jar(){
    count2=1
    for i in ${jar_name[*]}
    do
        echo -e "\033[44m $count2>   $i \033[0m"
        let ++count2
    done
    read -e -p '请输入更新的jar包:' num2
}

show_action(){
    count3=1
    for i in ${action_name[*]}
    do
        echo -e "\033[44m $count3>   $i \033[0m"
        let ++count3
    done
    read -e -p '请输入要执行的操作:' num3
}

copy_source(){
    echo -e "\033[42m 正在为$1分发资源$2... \033[0m"
    if [[ $2 == "activity-api" ||  $2  == "business-api" ]];then
         ansible $1qp-api -m copy -a "src=/projects/backend/global-pf/ims-$2.jar dest=/home/development/apps/"
    elif [[ $2 == "admin-api" || $2 == "game-betting" || $2 == "job-api" || $2 == "special-api" || $2 == "gateway" ]];then
         ansible $1qp-admin -m copy -a "src=/projects/backend/global-pf/ims-$2.jar  dest=/home/development/apps/"
    elif [[ $2 == "h5-admin" ]];then
         ansible $1qp-admin -m copy -a "src=/projects/imsfront/dist.zip dest=/home/development/front/"
    else
        continue
    fi
    echo -e "\033[42m $1资源分$2发完毕！ \033[0m"
}

update_source(){
    echo -e "\033[42m 正在为$1更新资源$2... \033[0m"
    if [[ $2 == "activity-api" ||  $2  == "business-api" ]];then
         ansible-playbook -e "host_list=$1qp-api" -v yml/$2.yml
    elif [[ $2 == "admin-api" || $2 == "game-betting" || $2 == "job-api" || $2 == "special-api"  || $2 == "gateway" ]];then
         ansible-playbook -e "host_list=$1qp-admin" -v yml/$2.yml
    elif [[ $2 == "h5-admin" ]];then
         ansible-playbook -e "host_list=$1qp-admin"  yml/h5-admin.yml
    else
        exit 1
    fi
    echo -e "\033[42m $1资源$2分发完毕！ \033[0m"
}

show_platForm
show_jar
echo ${platForm[num1-1]}
echo ${jar_name[num2-1]}
copy_source ${platForm[num1-1]} ${jar_name[num2-1]}
show_action
echo ${action_name[num3-1]}
if [[ ${action_name[num3-1]} == "exit_sys" ]];then
    exit 1
fi
read -e -p '是否继续？yes|no>:' confirm
if [[ $confirm == "yes" ]];then
    update_source ${platForm[num1-1]} ${jar_name[num2-1]} ${action_name[num3-1]}
fi
