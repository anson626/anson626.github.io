Docker 和 docker compose 安装

#install docker 
cd /etc/yum.repos.d/
wget https://download.docker.com/linux/centos/docker-ce.repo 
yum install docker-ce -y

systemctl start docker && systemctl enable docker && systemctl status docker

#install docker-compose
curl -L "https://github.com/docker/compose/releases/download/1.24.0/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose

chmod +x /usr/local/bin/docker-compose
docker-compose version

Docker Harbor

#登陆docker 先
docker login test.xinyuerh.com

#拉取镜像 需要docker login 先才能拉取
docker pull harbor.mir708090.com/edisguns/edis-guns-kubesphere@sha256:940786fe443a88d7758e2a068081f15af82ee35874c152c10bc7bce9c15f327a

#修改名
docker image tag 0e687ce6526b test.xinyuerh.com/edisguns/edis-guns-kubesphere:v1

#使用docker login 到
docker push test.xinyuerh.com/edisguns/edis-guns-kubesphere:v1
